    \end{description}

  \begin{textblock*}{11cm}(1cm,4.75cm)
    \begin{small}
    \begin{enumerate}[a]

      \item $\mathbf{dim}~\hom_0(D\setminus B_\omega)$ unknown.
    \end{enumerate}
